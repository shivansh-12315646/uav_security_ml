API Integration Guide
Python (Raspberry Pi / Companion Computer)
import requests, time, random

# Your drone's API key (from registration)
API_KEY = "YOUR_API_KEY_HERE"
BASE_URL = "http://YOUR_SERVER:8000"

while True:
    # Read real sensor data (replace with actual)
    telemetry = {
        "altitude": read_altimeter(),
        "speed": read_gps_speed(),
        "direction": read_compass(),
        "signal_strength": read_signal(),
        "distance_from_base": calc_distance(),
        "flight_time": get_flight_time(),
        "battery_level": read_battery(),
        "temperature": read_temp(),
        "vibration": read_imu(),
        "gps_accuracy": read_gps_hdop(),
    }

    resp = requests.post(
        f"{BASE_URL}/drone/api/telemetry/",
        json=telemetry,
        headers={"X-API-Key": API_KEY},
    )
    result = resp.json()
    print(f"Prediction: {result['prediction']}")

    time.sleep(5)  # Send every 5 seconds
cURL (Quick Test)
# Register a drone (requires login session)
curl -X POST http://localhost:8000/drone/api/register/ \
  -H "Content-Type: application/json" \
  -d '{"name": "DroneAlpha"}'

# Send telemetry data
curl -X POST http://localhost:8000/drone/api/telemetry/ \
  -H "Content-Type: application/json" \
  -H "X-API-Key: YOUR_API_KEY" \
  -d '{
    "altitude": 50,
    "speed": 10,
    "direction": 1,
    "signal_strength": 21,
    "distance_from_base": 3050,
    "flight_time": 123,
    "battery_level": 98,
    "temperature": 15,
    "vibration": 0,
    "gps_accuracy": 100
  }'

# Heartbeat check
curl http://localhost:8000/drone/api/heartbeat/ \
  -H "X-API-Key: YOUR_API_KEY"
